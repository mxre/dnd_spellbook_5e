\begin{spell}{Stoneskin}{PHB}{\nth{4} level abjuration}
{
    \spTime{1 action}
    \spRange{Touch}
    \spComponents{V, S, M (diamond dust worth 100 gp,
    which the spell consumes)}
    \spDuration{Concentration, up to 1 hour}
}
This spell turns the flesh of a willing creature you touch
as hard as stone. Until the spell ends, the target has
resistance to nonmagical bludgeoning, piercing, and
slashing damage.
\end{spell}
