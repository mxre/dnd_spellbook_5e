\begin{spell}{Fly}{PHB}{\nth{3} level transmutation}
{
    \spTime{1 action}
    \spRange{Touch}
    \spComponents{V, S, M (a wing feather from any bird)}
    \spDuration{Concentration, up to 10 minutes}
}
You touch a willing creature. The target gains a flying
speed of 60~feet for the duration. When the spell ends,
the target falls if it is still aloft, unless it can stop
the fall.

\note{At Higher Levels.} When you cast this spell using
a spell slot of \nth{4} level or higher, you can target one
additional creature for each slot level above \nth{3}.
\end{spell}
