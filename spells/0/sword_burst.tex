\begin{spell}{Sword Burst}{TCE}{Conjuration cantrip}
{
    \spTime{1 action}
    \spRange{Self (5-foot radius)}
    \spComponents{V}
    \spDuration{Instantaneous}
}
You create a momentary circle of spectral blades that
sweep around you. All other creatures within 5~feet of
you must each succeed on a Dexterity saving throw or take
1d6 force damage.

\note{At Higher Levels.} The spell's damage increases by
1d6 when you reach \nth{5} level (2d6), \nth{11} level (3d6),
and \nth{17} level (4d6).
\end{spell}
