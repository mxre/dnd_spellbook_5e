\begin{spell}{False Life}{PHB}{\nth{1} level necromancy}
{
    \spTime{1 action}
    \spRange{Self}
    \spComponents{V, S, M (a small amount of alcohol or distilled spirits)}
    \spDuration{1 hour}
}
Bolstering yourself with a necromantic facsimile of life,
you gain 1d4~+~4 temporary hit points for the duration.

\note{At Higher Levels.} When you cast this spell using a
spell slot of \nth{2} level or higher, you gain 5 additional
temporary hit points for each slot level above \nth{1}.
\end{spell}
