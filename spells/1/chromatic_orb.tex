\begin{spell}{Chromatic Orb}{PHB}{\nth{1} level evocation}
{
    \spTime{1 action}
    \spRange{90 feet}
    \spComponents{V, S, M (a diamond worth at least 50 gp)}
    \spDuration{Instantaneous}
}
You hurl a 4-inch-diameter sphere of energy at a creature
that you can see within range. You choose acid, cold, fire,
lightning, poison, or thunder for the type of orb you
create, and then make a ranged spell attack against the
target. If the attack hits, the creature takes 3d8 damage
of the type you chose.

\note{At Higher Levels.} When you cast this spell using a
spell slot of \nth{2} level or higher, the damage increases
by 1d8 for each slot level above \nth{1}.
\end{spell}
