\begin{spell}{Expedious Retreat}{PHB}{\nth{1} level transmutation}
{
    \spTime{1 bonus action}
    \spRange{Self}
    \spComponents{V, S}
    \spDuration{Concentration, up to 10 minutes}
}
This spell allows you to move at an incredible pace.
When you cast this spell, and then as a bonus action
on each of your turns until the spell ends, you can take
the Dash action.
\end{spell}
